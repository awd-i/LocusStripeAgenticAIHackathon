def _get_condition_description(self, wind: int, waves: float, visibility: float) -> str:
"""Generate human-readable condition description"""
conditions = []
if wind > 50:
conditions.append("hurricane-force winds")
elif wind > 35:
conditions.append("gale-force winds")
elif wind > 25:
conditions.append("strong winds")
if waves > 12:
conditions.append("very dangerous seas")
elif waves > 8:
conditions.append("rough seas")
elif waves > 4:
conditions.append("choppy waters")
if visibility < 1:
conditions.append("near-zero visibility")
elif visibility < 3:
conditions.append("poor visibility")
return ", ".join(conditions) if conditions else "moderate conditions"
